double PI = 3.14;
fn double circumference(double rad) {
  return 2.0*PI*rad*rad;
}
fn int shiftR(int a,int b) {
  return a>>b;
}
fn int shiftL(int a, int b) {
  return a<<b;
}
fn bool characterChar(char a, char b) {
    if(a : b) {
        return true;
    }
    else {
        return false;
    }
    return true;
}
fn bool characterString(string a, string b) {
    if(a : b) {
        return true;
    }
    else {
        return false;
    }
    return true;
}
